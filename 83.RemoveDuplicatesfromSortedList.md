## 83. Remove Duplicates from Sorted List
- 83. Remove Duplicates from Sorted List [Linked List] [Easy]

#### Tags
- [Linked List] [Easy]

#### Problem
Given a sorted linked list, delete all duplicates such that each element appear only once.

For example,  
Given 1->1->2, return 1->2.  
Given 1->1->2->3->3, return 1->2->3.

#### Data Structure
- Linked List

#### Algorithm/Method
- 

#### Input
- `ListNode* head`

#### Output
- `ListNode*`

#### Boudary Checking
- `if (head == NULL) return head;`

#### Solution #1
``` C++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* deleteDuplicates(ListNode* head) {
        if (head == NULL) return head;
        ListNode dummy(head->val + 1);
        dummy.next = head;
        recur(&dummy, head);
        return dummy.next;
    }
    
private:
    void recur(ListNode *prev, ListNode *cur) {
        if (cur == NULL) return;
        if (prev->val == cur->val) {
            prev->next = cur->next;
            delete cur;
            recur(prev, prev->next);
        } else {
            recur(prev->next, cur->next);
        }
    }
};
```

#### Notes
- Use Recurrence.

#### Solution #2
``` C++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode* deleteDuplicates(ListNode* head) {
        if (head == NULL) return head;
        for (ListNode *prev = head, *cur = head->next; 
             cur; cur = prev->next) {
            if (prev->val == cur->val) {
                prev->next = cur->next;
                delete cur;
            } else {
                prev = cur;
            }
        }
        return head;
    }
};
```

#### Notes
- Use iteration.
